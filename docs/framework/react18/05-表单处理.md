## 基本使用

```jsx
class Demo extends React.Component {
  state = {
    value: ''
  }

  handleChange = (e) => {
    this.setState({
      value: e.target.value
    })
  }

  render() {
    return (
      <>
        <input value={this.state.value} onChange={this.handleChange} />
      </>
    )
  }
}
```

## 多个表单元素

```jsx
class Demo extends React.Component {
  state = {
    username: '',
    email: ''
  }

  handleChange = (e) => {
    const name = e.target.name == 'username' ? 'username' : 'email'
    this.setState({
      [name]: e.target.value
    })
  }

  render() {
    const { username, email } = this.state
    return (
      <>
        <input value={username} name='username' onChange={this.handleChange} />
        <input value={email} name='email' onChange={this.handleChange} />
      </>
    )
  }
}
```

## 针对 select、checkbox、radio 的处理

```jsx title="select"
class Demo extends React.Component {
  state = {
    city: '北京'
  }

  change = (e) => {
    this.setState({
      city: e.target.value
    })
  }

  render() {
    const { city } = this.state
    return (
      <select value={city} onChange={this.change}>
        <option value="北京">北京</option>
        <option value="上海">上海</option>
        <option value="广州">广州</option>
      </select>
    )
  }
}
```

```jsx title="checkbox"
class Demo extends React.Component {
  state = {
    isChecked: false
  }

  change = (e) => {
    this.setState({
      isChecked: e.target.checked
    })
  }

  render() {
    return (
      <>
        <input type="checkbox" value="beijing" checked={this.state.isChecked} onChange={this.change} />
      </>
    )
  }
}
```

```jsx title="radio"
class Demo extends React.Component {
  state = {
    gender: 'male'
  }

  change = (e) => {
    this.setState({
      gender: e.target.value
    })
  }

  render() {
    const { gender } = this.state
    return (
      <>
        <input type="radio" value="male" checked={gender === 'male'} onChange={this.change} />
        <input type="radio" value="female" checked={gender === 'female'} onChange={this.change} />
      </>
    )
  }
}
```
