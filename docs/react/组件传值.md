## 父传子（props）

```jsx
// 父组件
class Parent extends React.Component {
  state = {
    msg: '父组件数据',
  };

  render() {
    return <Child msg={this.state.msg} />;
  }
}

// 子组件
class Child extends React.Component {
  render() {
    return <div>{this.props.msg}</div>;
  }
}
```

## 子传父（自定义事件）

```jsx
// 父组件
class Parent extends React.Component {
  state = {
    msg: '父组件数据',
  };

  fn = (msg) => {
    this.setState({ msg });
  };

  render() {
    return (
      <>
        <p>{this.state.msg}</p>
        <Child fn={this.fn} />
      </>
    );
  }
}

// 子组件
class Child extends React.Component {
  state = {
    msg: '子组件数据',
  };

  render() {
    return (
      <button onClick={() => this.props.fn(this.state.msg)}>按钮</button>
    );
  }
}
```
